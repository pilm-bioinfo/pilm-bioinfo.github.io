genes<-read.table("genes.tsv", header=F, row.names=1)

#Pre-process of expression matrix file with gene symbol annotation as row names

exp_genes<-merge(genes, exp, by=0, all=TRUE)

exp_genes<-exp_genes[,-c(1:2)]

exp_genes[1:3,1:3]

 

#Create a new SCESet object

sce <- newSCESet(countData=exp_genes)

 

#Load cell-cycle marker gene pairs

hs.pairs <- readRDS(system.file("exdata", "human_cycle_markers.rds", package="scran"))

#mm.pairs <- readRDS(system.file("exdata", "mouse_cycle_markers.rds", package="scran"))

 

#Load human genome-wide annotation file

source("https://bioconductor.org/biocLite.R")

biocLite("org.Hs.eg.db")  #install human genome-wide annotation

#biocLite("org.Mm.eg.db") #install mouse genome-wide annotation

library(org.Hs.eg.db)

#library(org.Mm.eg.db)

                          

#Prepare hs.pairs compatible ensembl annotation file 

anno <- select(org.Hs.eg.db, keys=rownames(sce), keytype="SYMBOL", column="ENSEMBL")

ensembl <- anno$ENSEMBL[match(rownames(sce), anno$SYMBOL)]

 

#Assign cell cycle phase for single cells

assignment <- cyclone(sce, hs.pairs, gene.names=ensembl)

 

#Process and save the output

phase_stat<-table(assignment$phases)

phase_single_cell<-assignment$phases

write.table(phase_single_cell, file="nsclc_5gex_cell_cycle.txt", quote=F, row.names=F, col.names=F)

 

 

#Practice2: t-SNE analysis of single cell RNA-Seq data

#Load needed libraries

#If you use a local computer, do the following

setwd("C:/Users/bioinformatics/Documents/10XG_workshop/")

#If you use the R-studio server, do the following

setwd("~/10XG_demo/")

 

#Load a single cell gene expression matrix file and a cell-cycle phase file

exp<-read.table("nsclc_5gex.csv",header=T,row.names=1,sep=",")

cell_cycle<-read.table("nsclc_5gex_cell_cycle.txt", header=F)

 

#Prepare a meta annotation with cell id and cell cycle phase information included for each individual cell

mymeta<-data.frame(id=rep("NSCLC",ncol(exp)),cycle=cell_cycle$V1)

rownames(mymeta)<-colnames(exp)

 

#Create a new Seurat object, and select genes with non-zero values in at least 5 single cells

nsclc <- CreateSeuratObject(raw.data = exp, project = "NSCLC", meta.data = mymeta, min.cells = 5)

 

#Pre-process of Seurat object (data normalization)

nsclc <- NormalizeData(nsclc)

nsclc <- FindVariableGenes(nsclc, do.plot = FALSE)

nsclc <- ScaleData(nsclc, display.progress = FALSE)

 

#Run PCA analysis to determine the number of low-dimensions for feature capture

nsclc <- RunPCA(nsclc, pcs.print = 0)

PCElbowPlot(nsclc)

 

#Identify clusters of cells by a shared nearest neighbor (SNN) modularity optimization based clustering algorithm

nsclc <- FindClusters(nsclc, dims.use = 1:13, print.output = FALSE)

nsclc <- RunTSNE(nsclc, dims.use = 1:13)

 

#Generate t-SNE plots

TSNEPlot(nsclc, do.label = FALSE, pt.size = 0.5, group.by = "id")

TSNEPlot(nsclc, do.label = FALSE, pt.size = 0.5, group.by = "cycle")

TSNEPlot(nsclc, do.label = T, do.return = T, pt.size = 0.5)
